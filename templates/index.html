// PARCELS try { const res = await fetch('./static/Parcels/parcels.geojson'); const data = await res.json(); parcelsLayer = L.geoJSON(data, { style: { color: '#444444', weight: 0.5, opacity: 0.7, fillColor: '#cccccc', fillOpacity: 0.05 }, onEachFeature: function(feature, layer) { const parcelNum = feature.properties?.parcel_num || ""; const owner = feature.properties?.deedholder || ""; const acres = feature.properties?.deeded_acr || ""; layer.bindPopup(` Parcel: ${parcelNum}
Owner: ${owner}
Acres: ${acres} `); if (parcelNum) { layer.bindTooltip(parcelNum, { permanent: true, direction: 'center', className: 'parcel-label' }); } } }); } catch (err) { console.log("Parcel layer failed:", err); }
